# -----------------------------
# SWITCH BETWEEN HPA MODES
# -----------------------------

# Namespaces
SOCIAL_NS=socialnetwork

# Files
export DEFAULT_HPA=deathstar-bench/hpa/default_hpa.yaml
export CONTEXT_HPA=deathstar-bench/hpa/context_aware_hpa_values.yaml
export SERVICE_VALUES=deathstar-bench/hpa/service_values.yaml
export ACTIVE_HPA="$CONTEXT_HPA"  # Will be updated when switching modes

# Option A: Switch to CONTEXT-AWARE HPA (uses external metrics via Prometheus Adapter)
# - Requires Prometheus Adapter installed and external metrics exposed
kubectl apply -f "$SERVICE_VALUES" -n "$SOCIAL_NS"
kubectl apply -f "$CONTEXT_HPA" -n "$SOCIAL_NS"
export ACTIVE_HPA="$CONTEXT_HPA"

# Verify
kubectl get hpa -n "$SOCIAL_NS"
kubectl describe hpa -n "$SOCIAL_NS" | sed -n '1,220p' | cat
# Expect: Metrics show external metric names like *_cpu_utilization_pct and ScalingActive=True

# Option B: Switch to DEFAULT HPA (CPU utilization % of requests)
kubectl apply -f "$DEFAULT_HPA" -n "$SOCIAL_NS"
export ACTIVE_HPA="$DEFAULT_HPA"

# Verify
kubectl get hpa -n "$SOCIAL_NS"
kubectl describe hpa -n "$SOCIAL_NS" | sed -n '1,220p' | cat
# Expect: Metrics show resource cpu (as a percentage of request)

# -----------------------------
# RESTART HPA (delete HPAs, scale to 0, re-apply previous, scale to 1)
# -----------------------------

# Delete existing HPAs in the socialnetwork namespace (ignore if not present)
kubectl delete hpa -n "$SOCIAL_NS" nginx-thrift-hpa compose-post-service-hpa text-service-hpa user-mention-service-hpa 2>/dev/null || true

# Scale all deployments to 0 replicas (pause workload)
kubectl get deploy -n "$SOCIAL_NS" -o name | xargs -r -n1 kubectl -n "$SOCIAL_NS" scale --replicas=0

# Re-apply service values and the previously active HPA manifest
kubectl apply -f "$SERVICE_VALUES" -n "$SOCIAL_NS"
kubectl apply -f "$ACTIVE_HPA" -n "$SOCIAL_NS"

# Scale all deployments back to 1 replica to restart with HPA attached
kubectl get deploy -n "$SOCIAL_NS" -o name | xargs -r -n1 kubectl -n "$SOCIAL_NS" scale --replicas=1

# Cleanup (optional): remove stale HPAs in the default namespace if present
kubectl delete hpa -n default nginx-thrift-hpa compose-post-service-hpa text-service-hpa user-mention-service-hpa 2>/dev/null || true
